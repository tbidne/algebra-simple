class (AGroup r, Hemiring r) => Rng r Source #

Defines a Rng i.e. a ring without multiplicative identity.

Examples:

Expand
  • \( 2 \mathbb{Z} \), the even integers.
>>> :{
  -- Addition
  f1 :: (Rng r) => r -> r
  f1 x = x .+. x
:}

>>> f1 5
10

>>> :{
  -- Zero
  f2 :: (Rng r) => r -> r
  f2 x = x .+. zero
:}

>>> f2 5
5

>>> :{
  -- Subtraction
  f3 :: (Rng r, Num r) => r -> r
  f3 x = x .-. 3
:}

>>> f3 5
2

>>> :{
  -- Multiplication
  f4 :: (Rng r) => r -> r
  f4 x = x .*. x
:}

>>> f4 5
25
