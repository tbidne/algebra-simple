class (Ring f, Semifield f) => Field f Source #

Defines a Field i.e. a structure that supports addition, subtraction, multiplication, and division.

Examples:

Expand
  • \( \mathbb{Q} \), the rationals.
>>> :{
  -- Addition
  f1 :: (Field k) => k -> k
  f1 x = x .+. x
:}

>>> f1 5
10

>>> :{
  -- Zero
  f2 :: (Field k) => k -> k
  f2 x = x .+. zero
:}

>>> f2 5
5

>>> :{
  -- Subtraction
  f3 :: (Field k, Num k) => k -> k
  f3 x = x .-. 3
:}

>>> f3 5
2

>>> :{
  -- Multiplication
  f4 :: (Field k) => k -> k
  f4 x = x .*. x
:}

>>> f4 5
25

>>> :{
  -- One
  f5 :: (Field k) => k -> k
  f5 x = x .*. one
:}

>>> f5 5
5

>>> :{
  -- Division
  f6 :: (Field k, Num k) => k -> k
  f6 x = x .%. 2
:}

f6 6 2
