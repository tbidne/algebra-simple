class (Demiring r, MGroup r) => Semifeld r Source #

Defines a Semifeld i.e. a semifield without additive identity.

Examples:

Expand
  • \( \mathbb{Q}^{\times} \), the positive rationals.
>>> :{
  -- Addition
  f1 :: (Semifeld k) => k -> k
  f1 x = x .+. x
:}

>>> f1 5
10

>>> :{
  -- Multiplication
  f2 :: (Semifeld k) => k -> k
  f2 x = x .*. x
:}

>>> f2 5
25

>>> :{
  -- One
  f3 :: (Semifeld k) => k -> k
  f3 x = x .*. one
:}

>>> f3 5
5

>>> :{
  -- Division
  f4 :: (Semifeld k, Num k) => k -> k
  f4 x = x .%. 2
:}

>>> f4 6
3
