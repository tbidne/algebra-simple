class AMonoid g => AGroup g where Source #

Defines an additive group.

Examples:

Expand
>>> :{
  -- Addition
  f1 :: (AGroup g) => g -> g
  f1 x = x .+. x
:}

>>> f1 5
10

>>> :{
  -- Zero
  f2 :: (AGroup g) => g -> g
  f2 x = x .+. zero
:}

>>> f2 5
5

>>> :{
  -- Subtraction
  f3 :: (AGroup g, Num g) => g -> g
  f3 x = x .-. 3
:}

>>> f3 5
2
