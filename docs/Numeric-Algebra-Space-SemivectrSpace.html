class (MSpace m r, Demimodule m r, Semifeld r) => SemivectrSpace m r | m -> r Source #

Defines a SemivectrSpace over a Semifeld.

Examples:

Expand
  • \( \mathbb{Q}^{\times} \times \mathbb{Q}^{\times} \), the two-dimensional positive rationals.
>>> :{
  -- Addition
  f1 :: (SemivectrSpace v k) => v -> v
  f1 v = v .+. v
:}

>>> f1 (8,4)
(16,8)

>>> :{
  -- Scalar multiplication
  f2 :: (SemivectrSpace v k, Num k) => v -> v
  f2 v = v .* 6
:}

>>> f2 (8,4)
(48,24)

>>> :{
  -- Scalar division
  f3 :: (SemivectrSpace v k, Num k) => v -> v
  f3 v = v .% 2
:}

>>> f3 (8,4)
(4,2)
