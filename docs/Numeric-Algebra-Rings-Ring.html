class (PseudoRing r, Semiring r) => Ring r Source #

Defines a Ring i.e. a structure that is an PseudoRing and a Semiring.

Examples:

Expand
  • \( \mathbb{Z} \), the integers.
>>> :{
  -- Addition
  f1 :: (Ring r) => r -> r
  f1 x = x .+. x
:}

>>> f1 5
10

>>> :{
  -- Zero
  f2 :: (Ring r) => r -> r
  f2 x = x .+. zero
:}

>>> f2 5
5

>>> :{
  -- Subtraction
  f3 :: (Ring r, Num r) => r -> r
  f3 x = x .-. 3
:}

>>> f3 5
2

>>> :{
  -- Multiplication
  f4 :: (Ring r) => r -> r
  f4 x = x .*. x
:}

>>> f4 5
25

>>> :{
  -- One
  f5 :: (Ring r) => r -> r
  f5 x = x .*. one
:}

>>> f5 5
5
